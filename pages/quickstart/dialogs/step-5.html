--- layout: quickstart permalink: /quickstart/dialogs/step-5/ --- {% capture markdown %} # Step 5: Add Edit Dialog In this step we're going to add an "edit" link to tweets that will launch a dialog to edit the text. ### Create Edit Link Run this command to create a component for our edit link: ``` lore generate component EditLink ``` Next update the content of the component to look like this. We're going to integrate an `onClick` handler that will launch and edit dialog when the link is clicked. Unlike the create dialog, we also need to pass in a `model` that contains the current data. This is required in order to populate the dialog with the exising data we want to edit. {% tabs tab1={"name": "ES5", "id": 101} | tab2={"name": "ES6", "id": 102} | tab3={"name": "ESNext", "id": 103} %} {% tab id=101, is_active=true %} ```jsx var React = require('react'); module.exports = React.createClass({ displayName: 'EditLink', propTypes: { tweet: React.PropTypes.object.isRequired }, onEdit: function() { var tweet = this.props.tweet; function updateTweet(params) { lore.actions.tweet.update(tweet, params); } lore.dialog.show(function() { return lore.dialogs.tweet.update({ model: tweet, onSubmit: updateTweet }); }); }, render: function() { return ( edit ); } }); ``` {% endtab %} {% tab id=102 %} ```jsx import React, { Component, PropTypes } from 'react'; class EditLink extends Component { constructor(props) { super(props); this.onEdit = this.onEdit.bind(this); } onEdit() { var tweet = this.props.tweet; function updateTweet(params) { lore.actions.tweet.update(tweet, params); } lore.dialog.show(function() { return lore.dialogs.tweet.update({ model: tweet, onSubmit: updateTweet }); }); } render() { return ( edit ); } } EditLink.propTypes = { tweet: PropTypes.object.isRequired }; export default EditLink; ``` {% endtab %} {% tab id=103 %} ```jsx import React, { Component, PropTypes } from 'react'; class EditLink extends Component { constructor(props) { super(props); this.onEdit = this.onEdit.bind(this); } static propTypes = { tweet: PropTypes.object.isRequired }; onEdit() { var tweet = this.props.tweet; function updateTweet(params) { lore.actions.tweet.update(tweet, params); } lore.dialog.show(function() { return lore.dialogs.tweet.update({ model: tweet, onSubmit: updateTweet }); }); } render() { return ( edit ); } } export default EditLink; ``` {% endtab %} {% endtabs %} ### Add an Edit Link to the Tweet Next we want to add the edit link to each tweet.
